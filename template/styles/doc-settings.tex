% --- Posición del encabezado -----------------------------------
% \topmargin controla la distancia desde el borde superior de la
% hoja hasta el encabezado. Valores más negativos = más arriba.
%
% Fórmula práctica:
%   distancia filo→logo = 1in(2.54cm) + \topmargin + \voffset
%
% Para tener 1cm entre el filo y el logo:
%   \topmargin = 1cm - 2.54cm = -1.54cm
%
% Ajusta SOLO la primera línea si quieres mover el encabezado:
%   más negativo  → encabezado más arriba
%   más positivo  → encabezado más abajo

\setlength{\headheight}{2.5cm}
\setlength{\headsep}{0.2cm}
\setlength{\topmargin}{-1.54cm}     % ← AJUSTA ESTE VALOR
\setlength{\textheight}{21cm}       % área de texto
\setlength{\footskip}{1cm}          % espacio para: nº página + línea + 2 líneas subtítulo

% ============================================================
% FORMATO DE CAPÍTULOS Y SECCIONES (estilo IFOP)
% ============================================================

% H1 (chapter): Verdana 13pt bold + regla azul debajo
\titleformat{\chapter}[block]
  {\verdanafont\fontsize{13}{16}\selectfont\bfseries}
  {\thechapter.\ }{0em}{}
  [\vspace{1pt}{\color{ifopazul}\rule{\linewidth}{1pt}}]
\titlespacing*{\chapter}{0pt}{0pt}{10pt}

% H2 (section): Verdana 13pt bold
\titleformat{\section}
  {\verdanafont\fontsize{13}{16}\selectfont\bfseries}
  {\thesection}{1em}{}
\titlespacing*{\section}{0pt}{14pt}{6pt}

% H3 (subsection): Verdana 13pt bold
\titleformat{\subsection}
  {\verdanafont\fontsize{13}{16}\selectfont\bfseries}
  {\thesubsection}{1em}{}
\titlespacing*{\subsection}{0pt}{10pt}{4pt}

% H4 (subsubsection): Verdana 13pt bold
\titleformat{\subsubsection}
  {\verdanafont\fontsize{13}{16}\selectfont\bfseries}
  {\thesubsubsection}{1em}{}
\titlespacing*{\subsubsection}{0pt}{8pt}{3pt}
